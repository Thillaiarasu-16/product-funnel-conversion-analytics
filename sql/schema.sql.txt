CREATE TABLE users (
    user_id INT PRIMARY KEY,
    signup_date DATE
);

CREATE TABLE events (
    event_id INT PRIMARY KEY,
    user_id INT,
    event_name VARCHAR(50),
    event_date DATE
);

CREATE TABLE payments (
    payment_id INT PRIMARY KEY,
    user_id INT,
    payment_date DATE,
    amount DECIMAL(10,2)
);
